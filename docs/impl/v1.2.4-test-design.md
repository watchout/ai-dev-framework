```ts
export interface CoverageGap {
  type: 'coverage' | 'environment' | 'tooling' | 'skill';

  resolution: string;
}

// src/lib/property/types.ts
export interface Invariant {
  id: string;            // INV-<FEATURE>-<NNN>
  description: string;
  property: () => boolean;
}

// src/lib/trace/types.ts
export interface TraceEntry {
  requirement: string;   // FR-XXX or AC-XXX

  code: string;          // file:line_range or symbol
}
```

## 3. シーケンス
- coverage gap: SPEC parse → 4 種 section 確認 → exit 0/2
- property: invariant 列挙 → fast-check 1000 runs → counterexample 記録
- trace: csv 読込 → 各 link を file/test/code で verify → exit 0/2

## 4. エラー処理 [推測]
- coverage section 不在 → exit 2
- property false → counterexample 記録 + test fail
- trace broken → exit 2、broken link list
